4. 寻找两个正序数组的中位数 Hard

题目描述

给定两个大小为 m 和 n 的正序（从小到大）数组 nums1 和 nums2。请你找出并返回这两个正序数组的中位数。

进阶

你能设计一个时间复杂度为 O(log (m+n)) 的算法解决此问题吗？

示例

示例 1：
输入：nums1 = [1,3], nums2 = [2]
输出：2.00000
解释：合并数组 = [1,2,3] ，中位数 2

示例 2：
输入：nums1 = [1,2], nums2 = [3,4]
输出：2.50000
解释：合并数组 = [1,2,3,4] ，中位数 (2 + 3) / 2 = 2.5

示例 3：
输入：nums1 = [0,0], nums2 = [0,0]
输出：0.00000

示例 4：
输入：nums1 = [], nums2 = [1]
输出：1.00000

示例 5：
输入：nums1 = [2], nums2 = []
输出：2.00000

提示

nums1.length == m
nums2.length == n
0 <= m <= 1000
0 <= n <= 1000
1 <= m + n <= 2000
-106 <= nums1[i], nums2[i] <= 106

求解

  • 二分查找法

    class Solution {
public:
    double findMedianSortedArrays(vector<int>& nums1, vector<int>& nums2) {
        int length = nums1.size() + nums2.size();
        if(length % 2 != 0)
            return getKthElement(nums1, nums2, length / 2 + 1);
        else
            return (getKthElement(nums1, nums2, length / 2) + getKthElement(nums1, nums2, length / 2 + 1)) / 2.0;
    }
    
    //得到第k小的元素
    int getKthElement(const vector<int> nums1, const vector<int> nums2, int k){
        int m = nums1.size();
        int n = nums2.size();
        int index1 = 0, index2 = 0; //分别用于检索nums1，nums2
        while(true){
            //边界情况
            if(index1 == m)
                return nums2[k + index2 - 1];
            if(index2 == n)
                return nums1[k + index1 - 1];
            if(k == 1)
                return min(nums1[index1], nums2[index2]);
            
            //一般情况
            int newIndex1 = min(index1 + k/2 - 1, m - 1);   //在防止数组越界的情况下进行二分查找
            int newIndex2 = min(index2 + k/2 -1, n - 1);
            int pivot1 = nums1[newIndex1];
            int pivot2 = nums2[newIndex2];
            if(pivot1 < pivot2){
                k -= newIndex1 - index1 + 1;    //减少k的个数
                index1 = newIndex1 + 1;
            }
            else{
                k -= newIndex2 - index2 + 1;
                index2 = newIndex2 + 1;
            }
        }
    }
};

  • 归并排序法

class Solution {
public:
    double findMedianSortedArrays(vector<int>& nums1, vector<int>& nums2) {
        int length = nums1.size() + nums2.size();
        double result = 0;
        if(length % 2 != 0)
            result = findNum(nums1, nums2, length/2);
        else
            result = (findNum(nums1, nums2, length/2) + findNum(nums1, nums2, length/2 - 1)) / 2;
        return result;        
    }

    //找到归并后数组的k号数
    double findNum(vector<int> nums1, vector<int>nums2, int k){
        int length = nums1.size() + nums2.size();
        vector<int> nums;
        int i = 0, j = 0;
        int cur = 0;
        while(i < nums1.size() && j < nums2.size() && cur <= k){
            if(nums1[i] < nums2[j]){
                nums.push_back(nums1[i++]);
                cur++;
            }
            else{
                nums.push_back(nums2[j++]);
                cur++;
            }
        }
        while(i < nums1.size() && cur <= k){
            nums.push_back(nums1[i++]);
            cur++;
        }
        while(j < nums2.size() && cur <= k){
            nums.push_back(nums2[j++]);
            cur++;
        }
        return nums[k];
    }
};

53. 最大子序和 Easy

题目描述

给定一个整数数组 nums ，找到一个具有最大和的连续子数组（子数组最少包含一个元素），返回其最大和。

示例

输入: [-2,1,-3,4,-1,2,1,-5,4]
输出: 6
解释: 连续子数组 [4,-1,2,1] 的和最大，为 6。

进阶

如果你已经实现复杂度为 O(n) 的解法，尝试使用更为精妙的分治法求解。

求解

  • 动态规划

    状态转移方程：令f(i)为以第i个数为结尾的最大子序和

    f(i) = max{ nums[i], f(i-1) + nums[i] }

    class Solution {
public:
    int maxSubArray(vector<int>& nums) {
        int pre = 0, maxNum = nums[0];
        for(int number : nums){
            pre = max(number, pre + number);
            maxNum = max(pre, maxNum);
        }
        return maxNum;
    }
};

  • 分治

    如果我们把区间 [0,n−1] 分治下去出现的所有子区间的信息都用堆式存储的方式记忆化下来，即建成一颗真正的树之后，我们就可以在 O(logn) 的时间内求到任意区间内的答案，我们甚至可以修改序列中的值，做一些简单的维护，之后仍然可以在 O(logn) 的时间内求到任意区间内的答案，对于大规模查询的情况下，这种方法的优势便体现了出来。

    class Solution {
public:
    struct status{
        int lSum;   //[l,r]内以l为左端点的最大子序和
        int rSum;   //[l,r]内以r为右端点的最大子序和
        int iSum;   //[l,r]内的区间和
        int mSum;   //[l,r]内的最大子序和
    };

    //治
    status pushUp(status lSub, status rSub){
        int iSum = lSub.iSum + rSub.iSum;
        int rSum = max(rSub.rSum, rSub.iSum + lSub.rSum);
        int lSum = max(lSub.lSum, lSub.iSum + rSub.lSum);
        int mSum = max(max(lSub.mSum, rSub.mSum), lSub.rSum + rSub.lSum);
        return (status){lSum, rSum, iSum, mSum};
    }

    //分
    status get(const vector<int> nums, int l, int r){
        if(l == r)
            return (status){nums[l], nums[l], nums[l], nums[l]};
        int mid = (l + r) / 2;
        status lSub = get(nums, l, mid);
        status rSub = get(nums, mid + 1, r);
        return pushUp(lSub, rSub);
    }

    int maxSubArray(vector<int>& nums) {
        return get(nums, 0, nums.size()-1).mSum;
    }
};

169. 多数元素 Easy

题目描述

给定一个大小为 n 的数组，找到其中的多数元素。多数元素是指在数组中出现次数大于⌊ n/2 ⌋的元素。

你可以假设数组是非空的，并且给定的数组总是存在多数元素。

示例

示例 1:
输入: [3,2,3]
输出: 3

示例 2:
输入: [2,2,1,1,1,2,2]
输出: 2

求解

  • 哈希表

    class Solution {
public:
    int majorityElement(vector<int>& nums) {
        unordered_map<int, int> counts;
        int majority, cnt = 0;
        for(int i : nums){
            counts[i]++;
            if(counts[i] > cnt){
                cnt = counts[i];
                majority = i;
            }
        }
        return majority;
    }
};

  • 分治

    class Solution {
public:
    int majorityElement(vector<int>& nums) {
        return majorityElementImp(nums, 0, nums.size()-1);    
    }

    int majorityElementImp(const vector<int> nums, int low, int high){
        if(low == high)
            return nums[low];
        int mid = (low + high) / 2;
        int leftMajority = majorityElementImp(nums, low, mid);
        int rightMajority = majorityElementImp(nums, mid+1, high);
        if(countInRange(nums, low, high, leftMajority) > (high-low+1)/2)
            return leftMajority;
        if(countInRange(nums, low, high, rightMajority) > (high-low+1)/2)
            return rightMajority;
        return -1;
    }

    int countInRange(const vector<int> nums, int low, int high, int majority){
        int count = 0;
        for(int i = low; i <= high; i++)
            if(nums[i] == majority)
                count++;
        return count;
    }
};

215. 数组中的第K个最大元素 Medium

题目描述

在未排序的数组中找到第 k 个最大的元素。请注意，你需要找的是数组排序后的第 k 个最大的元素，而不是第 k 个不同的元素。

示例

示例 1:
输入: [3,2,1,5,6,4] 和 k = 2
输出: 5
    
示例 2:
输入: [3,2,3,1,2,4,5,5,6] 和 k = 4
输出: 4

说明

你可以假设 k 总是有效的，且 1 ≤ k ≤ 数组的长度。

求解

  • 基于快速排序思想

    要找到第k个最大的数（即升序序列中的第n-k+1个数），因此其实我们无须关心其他数的排序情况。借鉴快排思想，每次选取一个基准元素pivot，然后进行数组划分，这样pivot最终的排序位置就确定下来了。

    设q为pivot对应数组下标，index为我们要找的第n-k+1个数对应的下标（n-k）

    ​ 若q = index，则nums[index]就是我们要找的数

    ​ 若q > index，则继续在nums[l, q-1]中找

    ​ 若q < index，则继续在nums[q+1, r]中找

    注意：此处与快速排序算法不同的是，这里每次在“分”之后的其中一个区间继续求解，而快速排序在“分”之后的每一个区间都要继续求解。因为这里只需求出第k个最大的数即可，不关心其他数，而快排是对所有数进行排序。

    时间复杂度：

    ​ 采用随机化方法，期望为：O(n)

    空间复杂度：

    ​ 即递归使用栈的空间，O(log n)

    class Solution {
public:
    
    int findKthLargest(vector<int>& nums, int k) {
        srand(time(0)); //用当前系统时间值设置种子
        return quickSelect(nums, 0, nums.size()-1, nums.size()-k); 
    }

    int quickSelect(vector<int>& nums, int l, int r, int index){
        int q = randomPartition(nums, l, r);
        if(q == index)
            return nums[q];
        else if(q > index)
            return quickSelect(nums, l, q-1, index);
        else
            return quickSelect(nums, q+1, r, index);
    }

    int randomPartition(vector<int>& nums, int l, int r){
        //random部分
        int rNum = rand() % (r-l+1) + l;    //随机化选择基准元素
        swap(nums[rNum], nums[r]);
        //partition部分
        int pivot = nums[r];
        int i = l - 1;
        for(int j = l; j <= r; j++){
            if(nums[j] < pivot){
                i++;
                swap(nums[i], nums[j]);
            }
        }
        //将pivot元素归位（即其最终位置）
        i++;    
        swap(nums[i], nums[r]);
        return i;
    }
};

  • 堆排序法

    构建一个大根堆，然后对堆顶元素（即最大的元素）进行k-1次删除和相应调整，第k次时对应的堆顶元素即为要找的数。

    建堆 -> 删除 -> 调整

    注意：此处手撕代码实现堆，若嫌麻烦可直接使用优先队列等容器。

    class Solution{
public:
    int findKthLargest(vector<int>& nums, int k){
        int heapSize = nums.size();
        buildMaxHeap(nums, heapSize);
        //删除 -> 调整
        for(int i = nums.size() - 1; i >= nums.size() - (k - 1); i--){
            swap(nums[0], nums[i]);
            heapSize--;
            maxHeapify(nums, 0, heapSize);
        }
        return nums[0];
    }

    //建堆
    void buildMaxHeap(vector<int>& nums, int heapSize){
        //对第一个以非叶子节点(heapSize/2 -1)为根节点的树按从下往上、从左往右的顺序进行调整
        for(int i = heapSize/2 - 1; i >=0; i--)
            maxHeapify(nums, i, heapSize);
    }

    //调整
    void maxHeapify(vector<int>& nums, int i, int heapSize){
        int lchild = 2 * i + 1;
        int rchild = 2 * i + 2;
        int largest = i;
        if(lchild < heapSize && nums[lchild] > nums[largest])
            largest = lchild;
        if(rchild < heapSize && nums[rchild] > nums[largest])
            largest = rchild;
        if(largest != i){
            swap(nums[i], nums[largest]);
            maxHeapify(nums, largest, heapSize);
        }
    }
};

240. 搜索二维矩阵 II Medium

题目描述

编写一个高效的算法来搜索 m x n 矩阵 matrix 中的一个目标值 target。

该矩阵具有以下特性：

  • 每行的元素从左到右升序排列
  • 每列的元素从上到下升序排列

示例

现有矩阵 matrix 如下：

[
  [1,   4,  7, 11, 15],
  [2,   5,  8, 12, 19],
  [3,   6,  9, 16, 22],
  [10, 13, 14, 17, 24],
  [18, 21, 23, 26, 30]
]

给定 target = 5，返回 true。

给定 target = 20，返回 false。

求解

利用该矩阵每行从左往右递增、每列从上往下递增的特点。

初始化一个指针(row, column)指向矩阵最左下角的数。

则

​ 若当前值大于目标值，则当前值左边的那些数可以排除，因此row--

​ 若当前值小于目标值，则当前值上面的那些数可以排除，因此column++

​ 若当前值等于目标值，则返回true

在指针迭代过程中，若发生越界现象，说明矩阵中无目标值，则返回false。

class Solution {
public:
    bool searchMatrix(vector<vector<int>>& matrix, int target) {
        if(matrix.empty()) //空数组时
            return false;
        int rowNum = matrix.size();
        int colNum = matrix[0].size();
        int row = rowNum - 1;
        int col = 0;
        while(row > -1 && col < colNum){
            if(matrix[row][col] == target)
                return true;
            else if(matrix[row][col] < target)
                col++;
            else
                row--;
        }    
        return false;
    }
};

241. 为运算表达式设计优先级 Medium

题目描述

给定一个含有数字和运算符的字符串，为表达式添加括号，改变其运算优先级以求出不同的结果。你需要给出所有可能的组合的结果。有效的运算符号包含 +,- 以及* 。

示例

示例 1:
输入: "2-1-1"
输出: [0, 2]
解释: 
((2-1)-1) = 0 
(2-(1-1)) = 2

示例 2:
输入: "2*3-4*5"
输出: [-34, -14, -10, -10, 10]
解释: 
(2*(3-(4*5))) = -34 
((2*3)-(4*5)) = -14 
((2*(3-4))*5) = -10 
(2*((3-4)*5)) = -10 
(((2*3)-4)*5) = 10

求解

  • 分治

    遍历字符串

    • 若当中存在运算符，以运算符作为分割线(相当于对其左、右部分分别加了括号)，求出其左、右两边相应可能的值，然后将左右两边的值根据运算符进行运算即可。

    • 若字符串只有数字，该数字就是最终的运算结果。

    同时，可用Hashmap来优化，避免对多个相同子问题的重复计算。

    class Solution {
public:
    unordered_map<string, vector<int>> resultMap;
    vector<int> diffWaysToCompute(string input) {
        //优化
        if(resultMap.find(input) != resultMap.end())
            return resultMap[input];
        vector<int> result;
        for(int i = 0; i < input.length(); i++){
            char ch = input[i];
            if(ch == '+' || ch =='-' || ch == '*'){
                vector<int> leftResult = diffWaysToCompute(input.substr(0, i));
                vector<int> rightResult = diffWaysToCompute(input.substr(i+1));
                for(int left : leftResult){
                    for(int right : rightResult){
                        if(ch == '+')
                            result.push_back(left + right);
                        if(ch == '-')
                            result.push_back(left - right);
                        if(ch == '*')
                            result.push_back(left * right);
                    }
                }
            }
        }
        //当字符串中只有数字时
        if(result.size() == 0)
            result.push_back(stoi(input));  //注意使用stoi将String转换为int
        resultMap.insert({input, result});
        return result;
    }
};

687. 最长同值路径 Medium

题目描述

给定一个二叉树，找到最长的路径，这个路径中的每个节点具有相同值。 这条路径可以经过也可以不经过根节点。

注意：两个节点之间的路径长度由它们之间的边数表示。

示例

示例 1:
输入:
         5
         / \
        4   5
       / \   \
      1   1   5
输出:
2

示例 2:
输入:
          1
         / \
        4   5
       / \   \
      4   4   5
输出:
2

注意: 给定的二叉树不超过10000个结点。 树的高度不超过1000。

求解

class Solution {
public:     
    int ans = 0;
    //以root为根节点的树中的最长同值路径
    int longestUnivaluePath(TreeNode* root) {
        longestPath(root);
        return ans;
    }
    //以node为起点的最长同值路径
    int longestPath(TreeNode* node){
        int maxLongestPath = 0;
        if(node == NULL)
            return 0;
        int right = longestPath(node->right);
        int left = longestPath(node->left);
        if(node->left != NULL && node->left->val == node->val && node->right != NULL && node->right->val == node->val )   //此时更新最长同值路径值ans（经过node、左子树、右子树）
            ans = max(ans, (right + left + 2));
        if(node->left != NULL && node->left->val == node->val)  //以node为起点左子树中最长同值路径
            maxLongestPath = left + 1;
        if(node->right != NULL && node->right->val == node->val) //以node为起点右子树中最长同值路径
            maxLongestPath = max(maxLongestPath, (right + 1));
        ans = max(ans, maxLongestPath); //更新ans
        return maxLongestPath;    //注意该函数求的是以node为起点的最长同值路径，所以返回的是maxLongestPath，而非ans
    }
};

698. 划分为k个相等的子集 Medium

题目描述

给定一个整数数组 nums 和一个正整数 k，找出是否有可能把这个数组分成 k 个非空子集，其总和都相等。

示例

示例 1：
输入： nums = [4, 3, 2, 3, 5, 2, 1], k = 4
输出： True

说明： 有可能将其分成 4 个子集（5），（1,4），（2,3），（2,3）等于总和。

提示

1 <= k <= len(nums) <= 16
0 < nums[i] < 10000

求解

class Solution {
public:
    bool canPartitionKSubsets(vector<int>& nums, int k) {
        int target = 0; //每个非空子集的总和
        for(int i = 0; i < nums.size(); i++)
            target += nums[i];
        if(target % k != 0)
            return false;
        target /= k;
        //将nums递减排序，从大到小地来试数，减少时间
        sort(nums.begin(), nums.end(), greater<int>());
        int n = nums.size();
        vector<int> sum;   //sum[i]表示第i个集合的总和
        
        return dfs(n, 0, sum, target, k, nums);
    }
    /*
    @para n 整数数组的元素数目； u 用于遍历nums； sum 存储集合总和的数组； target 每个子集的总和； k 非空子集数 
    */
    bool dfs(int n, int u, vector<int> sum, int target, int k, vector<int>& nums){
        if(u == n)  //所有元素已经放进了k个子集中
            return true;
        if(nums[u] > target) //有元素大于target，则该元素无法加入任何一个子集
            return false;
        for(int i = 0; i < sum.size(); i++){
            if(nums[u] + sum[i] > target)
                continue;
            sum[i] += nums[u];
            if(!dfs(n, u+1, sum, target, k, nums))   //若该元素的下一个nums中的元素无法加入任何一个子集，则回退一步，重新放置现在的元素
                sum[i] -= nums[u];
            else
                return true;
        }
        //第u元素无法加入现有子集，则新增一个子集
        if(sum.size() == k)
            return false;
        sum.push_back(nums[u]);
        if(!dfs(n, u+1, sum, target, k, nums)){ //看该元素的下一个元素是否可加入，是则返回真，否则回溯
            sum.pop_back();
            return false;
        }
        else
            return true;
    } 
};

932. 漂亮数组 Medium

题目描述

对于某些固定的 N，如果数组 A 是整数 1, 2, ..., N 组成的排列，使得：

对于每个 i < j，都不存在k 满足i < k < j使得 A[k] * 2 = A[i] + A[j]。

那么数组 A是漂亮数组。

给定 N，返回任意漂亮数组A（保证存在一个）。

示例

示例 1：
输入：4
输出：[2,1,4,3]

示例 2：
输入：5
输出：[3,1,2,5,4]

提示

1 <= N <= 1000

求解

性质：对漂亮数组进行线性变换后，仍然可以得到一个漂亮数组

因此，对于一个1...n的数组

分：将其分为一个值不超过(n+1)/2的数组和一个值不超过n/2的数组

解决：分别将其映射到1...n的所有奇数和偶数

治：将所得到的两部分（一部分是奇数，一部分是偶数）拼接起来即可

进一步优化：使用哈希表来存储已经求得的漂亮数组[m]，从而避免相同子问题的重复计算

class Solution {
public:

    unordered_map<int, vector<int>> resultMap;

    vector<int> beautifulArray(int N) {
        //优化
        if(resultMap.find(N) != resultMap.end())
            return resultMap[N];
        vector<int> result;
        if(N == 1)
            result.push_back(1);
        else{
            //分
            vector<int> odds = beautifulArray((N+1)/2); //用来映射奇数集合
            vector<int> evens = beautifulArray(N/2);    //用来映射偶数集合
            //治
            for(int odd : odds)
                result.push_back(2*odd - 1);
            for(int even : evens)
                result.push_back(2*even);
        }
        resultMap.insert({N, result});
        return result;
    }
};

 采用随机化方法，期望为：O(n)

空间复杂度：

 rpoints二维数组，O(n)

class Solution {
public:
    vector<vector<int>> rpoints;
    vector<vector<int>> kClosest(vector<vector<int>>& points, int K) {
        rpoints = points;
        srand(time(0));
        quickSelect(0, rpoints.size()-1, K-1);
        return vector<vector<int>>(rpoints.begin(), rpoints.begin()+K); //begin...begin+K
    }
    //快速选择
    void quickSelect(int l, int r, int index){
        int q = randomPartition(l, r);
        if(q == index)
            return;
        else if(q > index)
            quickSelect(l, q-1, index);
        else    
            quickSelect(q+1, r, index);
    }
    //随机划分
    int randomPartition(int l, int r){
        //random
        int rNum = rand() % (r-l+1) + l;
        swapPoints(r, rNum);
        //partition
        int pivot = dist(r);
        int i = l - 1;
        for(int j = l; j <= r; j++){
            if(dist(j) < pivot){
                i++;
                swapPoints(i, j);
            }
        }
        //基准元素归位（即其最终位置）
        i++;
        swapPoints(i, r);
        return i;
    }
    //交换两个点相应的值
    void swapPoints(int i, int j){
        int tmp1 = rpoints[i][0];
        int tmp2 = rpoints[i][1];
        rpoints[i][0] = rpoints[j][0];
        rpoints[i][1] = rpoints[j][1];
        rpoints[j][0] = tmp1;
        rpoints[j][1] = tmp2;
    }
    //计算点到原点的距离的平方
    int dist(int i){
        return rpoints[i][0]*rpoints[i][0] + rpoints[i][1]*rpoints[i][1];
    }
};

        }

  • 0 <= k <= 100000

跳水板问题有以下几种情况

  1. shorter = longer：此时长度只有1种，即k*shorter

  3. shorter < longer：此时长度有k+1种，即(k-i)*shorter + i*longer 其中0 <= i <=k

    ​ length = k*shorter + (longer - shorter)*i

    ​ 故length随i的变化而变化

    ​ i可取0...k，故长度有k+1种

class Solution {

            return vector<int> ();

            lengthOfBoard.push_back(k*shorter);
            return lengthOfBoard;
        }
        if(shorter < longer){
            for(int i = 0; i <= k; i++)
                lengthOfBoard.push_back((k-i)*shorter + i*longer);
        }
        return lengthOfBoard;
    }
};

            return INT_MIN;

        /*以中间元素mid为分界
        1. [left, mid]中最大连续数列和
        2. (mid, right]中最大连续数列和
        3. 经过mid的最大连续数列和
        取三者中最大值即可
        */
        if(left == right)
            return nums[left];
        int mid = (right + left) / 2;
        //情况1
        int leftSum = maxSum(nums, left, mid);
        //情况2
        int rightSum = maxSum(nums, mid + 1,  right);
        //情况3
        int leftMidSum = 0, leftMidMaxSum = INT_MIN;
        for(int i = mid; i >= left; i--){
            leftMidSum += nums[i];
            leftMidMaxSum = max(leftMidMaxSum, leftMidSum);
        }
        int rightMidSum = 0, rightMidMaxSum = INT_MIN;
        for(int i = mid + 1; i <= right; i++){
            rightMidSum += nums[i];
            rightMidMaxSum = max(rightMidMaxSum, rightMidSum);
        }
        int midSum = leftMidMaxSum + rightMidMaxSum;
        return max(max(leftSum, rightSum), midSum);
    }

        TreeNode *top, *pren = NULL, *head = NULL;  //栈顶指针 前一个结点指针 头指针

        }
